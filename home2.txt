            <?php
            function getWeeksDropdown($year = null)
            {
                global $startDateFrom; 
                global $startDateTo; 
                // If no year is provided, use the current year
                $year = $year ?? date('Y');

                // Calculate the number of weeks in the year
                $numWeeks = date('W', strtotime($year . '12-31'));

                // Calculate the last week of the year
                $lastWeek = $numWeeks;

                $selected_week = isset($_GET['week'])  ? $_GET['week'] : $lastWeek;

                // Create an array to store the week options
                $weeks = [];

                for ($week = 1; $week <= $numWeeks; $week++) {
                    // Calculate start and end dates for the week
                    $startDate = date('Y-m-d', strtotime($year . 'W' . sprintf('%02d', $week)));
                    $endDate = date('Y-m-d', strtotime($startDate . '+6 days'));

                    // Format the week option
                    $weekOption = "Week $week - " . date('F d', strtotime($startDate)) . " - " . date('F d, Y', strtotime($endDate));
                    $weeks[$week] = $weekOption;
                    if($week == $selected_week){
                        $startDateFrom = $startDate;
                        $startDateTo = $endDate;
                    }
                }

                // Create the dropdown
                echo '<select  style="width: 300px"  id="week-select" class="form-control pull-right" name="week">';
                foreach ($weeks as $week => $weekOption) {
                    $selected = ($week == $selected_week) ? 'selected' : '';
                    echo '<option value="' . $week . '" ' . $selected . '>' . $weekOption . '</option>';
                }
                echo '</select>';
            }
            getWeeksDropdown();
            ?>
            </form>
